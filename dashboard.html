// Membuat peta Indonesia var map = L.map('map') .setView([-2.5,118],5); // OpenStreetMap var osm = L.tileLayer( 'https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', { maxZoom:19, attribution:'OpenStreetMap' } ); // ESRI Satellite var satellite = L.tileLayer( 'https://server.arcgisonline.com/ArcGIS/rest/services/World_Imagery/MapServer/tile/{z}/{y}/{x}', { attribution:'ESRI Satellite' } ); // Default Layer osm.addTo(map); // Basemap Control var baseMaps = { "OpenStreetMap": osm, "Satellite": satellite }; L.control.layers(baseMaps).addTo(map); // Marker Indonesia L.marker([-6.2,106.8]) .addTo(map) .bindPopup( "Jakarta
Pusat Monitoring" );
